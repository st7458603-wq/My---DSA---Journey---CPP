
#include <iostream>
using namespace std;

// Function to find number of ways
int friendsPairing(int n) {

    // Base cases
    if (n == 1 || n == 2) {
        return n;
    }

    // Single
    int single = friendsPairing(n - 1);

    // Pair
    int pairUp = (n - 1) * friendsPairing(n - 2);

    // Total ways
    return single + pairUp;
}

int main() {

    int n;

    cout << "Enter number of friends: ";
    cin >> n;

    cout << "Total ways: " << friendsPairing(n);

    return 0;
}
